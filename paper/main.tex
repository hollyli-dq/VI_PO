%  ABSTRACT

\begin{abstract}
Partial orders arise in two distinct yet interconnected settings:
(i)~\emph{trace de-linearization}, where a latent poset governs noisy sequential
observations, and (ii)~\emph{constrained structure learning}, where domain experts
supply ordering priors over variables.
Existing approaches treat these settings separately---probabilistic poset models
rely on expensive MCMC, while differentiable DAG learners lack scalable mechanisms
for general partial-order priors.
We present \textbf{Variational Partial Orders (VPO)}, a unified framework with
three contributions:
\textbf{(1)}~a latent-embedding poset model with a frontier-restricted
Plackett--Luce likelihood for recovering partial orders from traces;
\textbf{(2)}~importance-weighted variational inference (IWAE) with a
VIMCO-$\star$ gradient estimator that achieves $\sqrt{N}$
signal-to-noise scaling and avoids reparameterization through the
discrete dominance map;
and \textbf{(3)}~an efficient encoding of partial-order priors via transitive
reduction and maximal-path factoring, reducing constraint cost from
$\Theta(m^2)$ to $O(m)$ for chain-like orders.
We prove that enforcing cover relations alone suffices for dominance-based
embeddings (Theorem~\ref{thm:cover}), enabling compact injection of ordering
priors into both variational poset inference and differentiable causal discovery.
Experiments on synthetic posets, scaling stress tests, and workflow-log benchmarks
demonstrate that VPO recovers latent orders with calibrated uncertainty, scales
gracefully, and improves downstream structure learning when used as an ordering prior.
\end{abstract}

\section{Introduction}
\label{sec:intro}
% ==================================================================

Partial orders encode precedence and concurrency among a set of entities.
They arise naturally whenever observed sequences are noisy linearizations of
an underlying dependency structure: workflow execution
logs~\citep{vanderaalst2016process}, tool-augmented LLM
traces~\citep{yao2023react,schick2024toolformer}, robotic assembly plans,
and treatment protocols.
In a separate but related setting, causal discovery methods seek directed
acyclic graphs (DAGs) that explain observational data, and domain experts
frequently possess partial-order knowledge---e.g., gene $A$ activates before
gene $B$---that should constrain the search
space~\citep{pearl2009causality,spirtes2001causation}.

Despite sharing the same mathematical object, these two regimes have developed
largely disjoint toolkits.
\emph{Trace models} typically treat the poset as a discrete latent variable and
resort to Markov chain Monte Carlo (MCMC), which scales poorly in the number
of actions $m$.
\emph{Differentiable DAG learners}~\citep{zheng2018notears,bello2022dagma}
operate in continuous parameter spaces and historically lacked efficient
mechanisms for incorporating general partial-order priors.
Recent work by \citet{ban2024differentiable} shows that the na\"ive approach
of enumerating all pairs in the transitive closure $O^+$ incurs
$\Theta(m^2)$ constraint terms for a chain of length $m$, and proposes an
augmented-acyclicity alternative.

\paragraph{Contributions.}
We introduce \textbf{Variational Partial Orders (VPO)}, a framework that
bridges trace-based poset inference and order-constrained structure learning:

\begin{enumerate}[leftmargin=*,topsep=2pt,itemsep=1pt]
\item \textbf{Latent-embedding poset model (Section~\ref{sec:model}).}
  We formalize a generative model in which latent embeddings
  $U\in\R^{m\times K}$ induce a strict partial order via coordinate-wise
  dominance, and traces are generated through a frontier-restricted
  Plackett--Luce likelihood with trembling-hand noise.
\item \textbf{Importance-weighted VI with VIMCO-$\star$ gradients
  (Section~\ref{sec:vi}).}
  Because the dominance map $h(U)=\Dom(U)$ is piecewise-constant,
  reparameterized gradients are zero almost everywhere.
  We derive the score-function gradient of the VR-IWAE
  objective~\citep{burda2016iwae,daudel2026importance} specialized to our
  model, and adopt the VIMCO-$\star$ control variate~\citep{daudel2026importance}
  whose signal-to-noise ratio (SNR) scales as $\sqrt{N}$.
\item \textbf{Efficient partial-order priors via cover relations
  (Sections~\ref{sec:podsl}--\ref{sec:unified}).}
  We prove (Theorem~\ref{thm:cover}) that enforcing dominance on the
  transitive reduction $O^-$ automatically propagates to the full closure
  $O^+$, reducing the number of constraints from $\Theta(m^2)$ to $|O^-|$
  (often $O(m)$) and connecting to the maximal-path factoring
  of~\citet{ban2024differentiable}.
\end{enumerate}

\section{Background}
\label{sec:background}
% ==================================================================

\subsection{Partial Orders, Transitive Closure, and Transitive Reduction}
A \emph{strict partial order} on a finite set $[m]=\{1,\dots,m\}$ is an
irreflexive, transitive binary relation $\succ$.
Its \emph{transitive closure} $O^+$ contains all pairs implied by
transitivity; its \emph{transitive reduction} $O^-$ is the unique smallest
relation satisfying $(O^-)^+=O^+$~\citep{davey2002introduction}.

\begin{remark}[Quadratic blow-up]
\label{rem:blowup}
For a total order (chain) $v_1\succ\cdots\succ v_m$,
$|O^+|=\binom{m}{2}=\Theta(m^2)$ while $|O^-|=m-1$.
Any algorithm that enumerates $O^+$ inherits this quadratic cost.
\end{remark}

\subsection{Plackett--Luce Models and Linear Extensions}
The Plackett--Luce (PL) model~\citep{plackett1975luce,luce1959individual}
defines a distribution over permutations via sequential softmax choices.
A \emph{linear extension} of a poset $h$ is a total order consistent with
$h$.
Standard PL does not enforce poset consistency; we introduce a
\emph{frontier-restricted} variant in Section~\ref{sec:likelihood}.

\subsection{Importance-Weighted Variational Inference}
Given a latent-variable model $p(x,z)$ and variational family $q_\phi(z)$,
the importance-weighted autoencoder (IWAE)
bound~\citep{burda2016iwae} with $N$ particles is
$\mathcal{L}_N(\phi)=\E[\log\frac{1}{N}\sum_{n=1}^N w(z_n)]$,
where $w(z)=p(x,z)/q_\phi(z)$.
This tightens monotonically in $N$~\citep{burda2016iwae}.
When reparameterization is unavailable, score-function
estimators~\citep{williams1992reinforce} are needed; the VIMCO
family~\citep{mnih2016vimco} uses leave-one-out baselines to reduce
variance.
\citet{daudel2026importance} generalize this to a VR-IWAE family
parameterized by $\alpha\in[0,1)$ and introduce the VIMCO-$\star$
estimator with $\sqrt{N}$ SNR scaling.

\subsection{Differentiable DAG Learning}
NOTEARS~\citep{zheng2018notears} casts structure learning as continuous
optimization with a smooth acyclicity constraint $h(W)=\tr(e^{W\circ W})-d=0$.
DAGMA~\citep{bello2022dagma} replaces this with a log-determinant characterization.
\citet{ban2024differentiable} extend both to handle partial-order priors
via augmented acyclicity over maximal paths.

\section{Latent Posets from Traces}
\label{sec:model}
% ==================================================================

\subsection{Latent Embeddings and the Dominance Map}
\label{sec:dominance}
Let $\mathcal{A}=\{1,\dots,m\}$ denote a set of actions and
$K\in\N$ the latent embedding dimension.
We assign each action a latent vector and place a Gaussian prior:
\begin{equation}
U_j \sim \mathcal{N}(0,\,\Sigma_\rho),\quad j=1,\dots,m,
\label{eq:prior_U}
\end{equation}
where $\Sigma_\rho=(1-\rho)I_K+\rho\mathbf{1}\mathbf{1}^\top$ is an
exchangeable covariance with correlation $\rho\in[0,1)$.

\begin{definition}[Coordinate-wise dominance]
\label{def:dominance}
The dominance map $\Dom:\R^{m\times K}\to\{0,1\}^{m\times m}$ is defined by
$i\succ_{\Dom(U)}j$ iff $U_{i,k}>U_{j,k}$ for all $k\in[K]$.
We write $h=h(U)\coloneqq\Dom(U)$ for the induced relation.
\end{definition}

\begin{theorem}[Dominance induces a strict partial order]
\label{thm:spo}
For any $U\in\R^{m\times K}$, the relation $\succ_{h(U)}$ is irreflexive
and transitive, hence $h(U)$ is a strict partial order on $\mathcal{A}$.
\end{theorem}
\begin{proof}
Irreflexivity: $U_{i,k}>U_{i,k}$ is a contradiction for any $k$.
Transitivity: if $U_{i,k}>U_{j,k}$ and $U_{j,k}>U_{l,k}$ for all $k$,
then $U_{i,k}>U_{l,k}$ for all $k$ by transitivity of $>$ in $\R$.
\end{proof}

The embedding dimension $K$ controls poset complexity: $K=1$ yields a
total order (chain); as $K$ grows, more pairs become incomparable,
producing sparser posets.

\subsection{Frontier-Restricted Plackett--Luce Likelihood}
\label{sec:likelihood}
Given a trace $y=(y_1,\dots,y_T)$ and history $y_{<t}=(y_1,\dots,y_{t-1})$,
define the remaining actions $R_t\coloneqq\mathcal{A}\setminus\{y_1,\dots,y_{t-1}\}$
and the \emph{frontier}:
\begin{equation}
F_t(h;\,y_{<t})
\coloneqq \bigl\{a\in R_t : \nexists\, b\in R_t\text{ s.t.\ }
b\succ_h a\bigr\}.
\label{eq:frontier}
\end{equation}
The frontier comprises the minimal elements of $R_t$ under $h$---exactly
the actions whose predecessors have all been executed.

We model each step via a frontier-restricted Plackett--Luce distribution
with trembling-hand noise $\epsilon\in(0,1)$:
\begin{equation}
p(y_t\mid y_{<t},h,\beta,\epsilon)
= (1-\epsilon)\,
\frac{\exp\bigl(\beta\, Q(y_t;h,t)\bigr)}
     {\sum_{a\in F_t(h)}\exp\bigl(\beta\, Q(a;h,t)\bigr)}\,
     \indic[y_t\in F_t(h)]
\;+\;
\epsilon\,\frac{1}{|R_t|},
\label{eq:step_likelihood}
\end{equation}
where $Q(a;h,t)$ is a utility function (set to zero unless additional
action features are available), $\beta>0$ is an inverse temperature, and
the indicator ensures the softmax term contributes only when $y_t$ is
feasible.  The full trace likelihood factorizes as:
\begin{equation}
p(y\mid h,\beta,\epsilon) = \prod_{t=1}^T p(y_t\mid y_{<t},h,\beta,\epsilon).
\label{eq:trace_ll}
\end{equation}

The trembling-hand term assigns positive probability to any action in
$R_t$, ensuring the likelihood is well-defined even when $y_t\notin F_t(h)$.
This is critical for VI: without it, many particles would receive
$\log p=-\infty$, collapsing the importance-weighted objective.

\subsection{Posterior}
\label{sec:posterior}
Given a dataset $D=\{y^{(i)}\}_{i=1}^n$ of traces, the posterior over
latents $z\coloneqq(U,\rho,\beta)$ (fixing $K$ for now) is:
\begin{equation}
p(z\mid D)
\propto p(U\mid\rho)\,p(\rho)\,p(\beta)\,
\prod_{i=1}^n p\bigl(y^{(i)}\mid h(U),\beta,\epsilon\bigr).
\label{eq:posterior}
\end{equation}
We place $p(\rho)=\mathrm{Beta}(a_\rho,b_\rho)$ and
$p(\beta)=\mathrm{LogNormal}(\mu_\beta,\sigma_\beta^2)$ as priors.
The posterior is intractable due to the combinatorial nature of $h(U)$;
we approximate it via importance-weighted VI.

\section{Importance-Weighted Variational Inference for Latent Posets}
\label{sec:vi}
% ==================================================================

\subsection{The ELBO and Why Reparameterization Fails}
\label{sec:elbo}
Let $q_\phi(z)$ be a variational family.
The evidence lower bound (ELBO) is:
\begin{align}
\mathcal{L}(\phi)
&= \E_{q_\phi}\!\left[
\sum_{i=1}^n \log p(y^{(i)}\mid h(U),\beta,\epsilon)
+ \log p(U\mid\rho) + \log p(\rho) + \log p(\beta)
- \log q_\phi(z)
\right].
\label{eq:elbo}
\end{align}
The dominance map $h(U)=\Dom(U)$ is \emph{piecewise constant}:
within any open region of the embedding space the induced poset is fixed,
and $h$ is discontinuous at region boundaries.
Consequently, the pathwise derivative
$\nabla_U\log p(y\mid h(U),\beta,\epsilon)=0$ almost everywhere,
and standard reparameterization
tricks~\citep{kingma2014vae,rezende2014stochastic} produce zero-gradient
estimates for the likelihood term.
We therefore require score-function methods.

\subsection{VR-IWAE Objective}
\label{sec:vr_iwae}
Draw $z_{1:N}\stackrel{iid}{\sim}q_\phi$ and define
the unnormalized importance weight:
\begin{equation}
w(z_n) \coloneqq \frac{p(D,z_n)}{q_\phi(z_n)},
\qquad
\log p(D,z_n)
= \log p(z_n) + \sum_{i=1}^n \log p(y^{(i)}\mid h(U_n),\beta_n,\epsilon).
\label{eq:log_weight}
\end{equation}
The VR-IWAE family~\citep{daudel2026importance}, parameterized by
$\alpha\in[0,1)$, is:
\begin{equation}
\mathcal{L}^{(\alpha)}_N(\phi)
= \frac{1}{1-\alpha}\,\E\!\left[
\log\!\left(\frac{1}{N}\sum_{n=1}^N w(z_n)^{1-\alpha}\right)
\right].
\label{eq:vr_iwae}
\end{equation}
Setting $\alpha=0$ recovers the standard IWAE
bound~\citep{burda2016iwae}, which tightens monotonically in $N$.
Setting $\alpha>0$ interpolates toward the ELBO and corresponds
to R\'{e}nyi-divergence
minimization~\citep{li2016renyi,daudel2026importance}.

\subsection{Score-Function Gradient Decomposition}
\label{sec:sf_grad}
Define the self-normalized weights:
\begin{equation}
\bar{W}_n^{(1-\alpha)}
\coloneqq
\frac{w(z_n)^{1-\alpha}}{\sum_{j=1}^N w(z_j)^{1-\alpha}}.
\label{eq:snw}
\end{equation}
For any parameter block $\psi\subseteq\phi$, the exact gradient
decomposes as~\citep{daudel2026importance}:
\begin{align}
\nabla_\psi\mathcal{L}^{(\alpha)}_N
&= \E_{z_{1:N}}\!\Bigg[
\underbrace{%
  \sum_{n=1}^N \bar{W}_n^{(1-\alpha)}\,\nabla_\psi\log w(z_n)
}_{\text{self-normalized IS term}}
\;+\;
\underbrace{%
  \frac{1}{1\!-\!\alpha}
  \Bigl(\sum_{n=1}^N\nabla_\psi\log q_\phi(z_n)\Bigr)
  \log\!\Bigl(\tfrac{1}{N}\textstyle\sum_{j=1}^N w(z_j)^{1-\alpha}\Bigr)
}_{\text{REINFORCE term (high variance)}}
\Bigg].
\label{eq:reinforce_grad}
\end{align}
Since $p(D,z)$ is independent of $\phi$, we have
$\nabla_\psi\log w(z)=-\nabla_\psi\log q_\phi(z)$.
The first term is well-conditioned, but the second
(REINFORCE) term exhibits SNR collapse as $N$
grows~\citep{daudel2026importance,mnih2016vimco}, motivating
variance-reduction techniques.

\subsection{VIMCO-$\star$ Estimator}
\label{sec:vimco_star}

The VIMCO framework~\citep{mnih2016vimco} replaces the global REINFORCE
signal with per-sample leave-one-out baselines $f_{-n}$ depending only on
$z_{-n}=(z_1,\dots,z_{n-1},z_{n+1},\dots,z_N)$; since
$\E[\nabla_\psi\log q_\phi(z_n)\cdot f(z_{-n})]=0$, unbiasedness is
preserved while variance is reduced.
\citet{daudel2026importance} provide a systematic family of such
baselines for general $\alpha$ and, for the IWAE case $\alpha=0$,
derive a particularly simple form.

\begin{proposition}[VIMCO-$\star$ estimator, {\citet[Eq.~15]{daudel2026importance}}]
\label{prop:vimco_star}
For $\alpha=0$, let $W_n=w(z_n)/\sum_j w(z_j)$.
The VIMCO-$\star$ gradient estimator is:
\begin{equation}
\boxed{%
\widehat{\nabla_\psi\mathcal{L}^{(0)}_N}
= \sum_{n=1}^N \Bigl[
W_n\,\nabla_\psi\log w(z_n)
\;-\;
\log(1-W_n)\,\nabla_\psi\log q_\phi(z_n)
\Bigr].
}
\label{eq:vimco_star}
\end{equation}
\end{proposition}

For variational parameters $\psi\subseteq\phi$
(where $\nabla_\psi\log w=-\nabla_\psi\log q$),
Eq.~\eqref{eq:vimco_star} simplifies to:
\begin{equation}
\widehat{\nabla_\psi\mathcal{L}^{(0)}_N}
= -\sum_{n=1}^N
\bigl(W_n+\log(1-W_n)\bigr)\,\nabla_\psi\log q_\phi(z_n).
\label{eq:vimco_star_simple}
\end{equation}

\begin{remark}[SNR scaling]
\citet{daudel2026importance} prove that the SNR of the
VIMCO-$\star$ estimator scales as $\Theta(\sqrt{N})$, in contrast to
$O(1/\sqrt{N})$ decay for naive multi-sample
REINFORCE~\citep{williams1992reinforce}.
\end{remark}

\begin{remark}[No gradients through $\Dom(U)$]
\label{rem:no_grad}
Equations~\eqref{eq:vimco_star}--\eqref{eq:vimco_star_simple} require only
$\log q_\phi(z_n)$ and its gradient---\emph{not} gradients through the
dominance map.
This is the key practical advantage for latent-poset inference.
\end{remark}

\subsection{Variational Family}
\label{sec:var_family}
We use a mean-field Gaussian approximation:
\begin{equation}
q_\phi(z) = q_\phi(U)\,q_\phi(\rho)\,q_\phi(\beta),
\qquad
q_\phi(U) = \prod_{j=1}^m \mathcal{N}(U_j;\,\mu_j,\,\mathrm{diag}(\sigma_j^2)),
\label{eq:var_family}
\end{equation}
with $q_\phi(\beta)=\mathrm{LogNormal}$ and $q_\phi(\rho)=\mathrm{Beta}$.
The variational parameters are
$\phi=\{\mu_j,\log\sigma_j\}_{j=1}^m\cup\{\phi_\beta,\phi_\rho\}$.
Extensions to structured (e.g., low-rank) covariance are straightforward
but not explored here.

\subsection{Algorithm}

\begin{algorithm}[t]
\caption{VPO-VI: Importance-weighted VI for latent posets with
VIMCO-$\star$}
\label{alg:vpo_vi}
\KwIn{Traces $D=\{y^{(i)}\}_{i=1}^n$, particles $N$, learning rate
$\eta$, slip $\epsilon$}
Initialize variational parameters $\phi$\;
\While{\textnormal{not converged}}{
  Sample $z_{1:N}\sim q_\phi(z)$\;
  \For{$n=1$ \KwTo $N$}{
    $h^{(n)}\leftarrow\Dom(U^{(n)})$
    \tcp*[f]{$O(m^2 K)$}
    $\log p_n \leftarrow \log p(z_n)
      + \sum_{i=1}^n\log p(y^{(i)}\mid h^{(n)},\beta^{(n)},\epsilon)$\;
    $\log q_n \leftarrow \log q_\phi(z_n)$\;
    $\log w_n \leftarrow \log p_n - \log q_n$\;
  }
  $W_n \leftarrow \exp(\log w_n)/\!\sum_{j}\exp(\log w_j)$
  \tcp*[f]{log-space softmax}
  $\widehat{\nabla_\phi}\leftarrow
    -\sum_{n=1}^N\bigl(W_n+\log(1\!-\!W_n)\bigr)\,
    \nabla_\phi\log q_\phi(z_n)$
  \tcp*[f]{Eq.~\eqref{eq:vimco_star_simple}}
  $\phi\leftarrow\phi+\eta\,\widehat{\nabla_\phi}$
  \tcp*[f]{Adam~\citep{kingma2014adam}}
}
\KwOut{Posterior $q_\phi(z)$; poset samples $h^{(n)}=\Dom(U^{(n)})$}
\end{algorithm}

\section{Partial-Order-Constrained Differentiable Structure Learning}
\label{sec:podsl}
% ==================================================================

\subsection{Path-Prohibition Penalty and Quadratic Overhead}
In differentiable DAG learning, an adjacency matrix $W\in\R^{d\times d}$
parameterizes a weighted directed graph.
Acyclicity is enforced via a smooth constraint
$h(W)=\tr(e^{W\circ W})-d=0$~\citep{zheng2018notears}.
To incorporate a partial-order prior $O$, one can prohibit paths
that violate $O$: for each $(i,j)\in O^+$, forbid $j\rightsquigarrow i$.
A continuous relaxation~\citep{ban2024differentiable} yields:
\begin{equation}
\mathcal{P}(W,O)
= \sum_{(i,j)\in O^+}\Bigl(\sum_{l=1}^d (W\circ W)^l\Bigr)_{j,i},
\label{eq:naive_penalty}
\end{equation}
whose cost scales with $|O^+|$---i.e., $\Theta(m^2)$ for chains
(Remark~\ref{rem:blowup}).

\subsection{Efficient Encoding via Augmented Acyclicity}
\citet{ban2024differentiable} avoid the transitive-closure enumeration
by working with the transitive reduction $O^-$ and its maximal paths
$\mathcal{P}(O^-)$.
Define the augmented graph:
\begin{equation}
A(W,o) = W + \tau W_o - W\circ W_o,
\qquad
(W_o)_{ij} = \indic[(i,j)\in o],
\label{eq:augment}
\end{equation}
where $\tau>0$ and $o$ is a directed path.
The augmented-acyclicity function is:
\begin{equation}
h'(W,O) = \sum_{o\in\mathcal{P}(O^-)} h\bigl(A(W,o)\bigr).
\label{eq:h_prime}
\end{equation}

\begin{theorem}[Equivalence, {\citet[Theorem~1]{ban2024differentiable}}]
\label{thm:equiv}
$G(W)$ is a DAG satisfying all constraints in $O$ if and only if
$h'(W,O)=0$.
\end{theorem}

\begin{algorithm}[t]
\caption{PO-DSL: Augmented acyclicity for partial-order
priors~\citep{ban2024differentiable}}
\label{alg:po_dsl}
\KwIn{Adjacency $W$, partial order $O$, acyclicity function $h$,
strength $\tau$}
$O^-\leftarrow\textsc{TransitiveReduction}(O)$\;
$\mathcal{P}\leftarrow\textsc{MaximalPaths}(O^-)$\;
$s\leftarrow 0$\;
\For{each path $o\in\mathcal{P}$}{
  $W_o\leftarrow$ indicator matrix for edges in $o$\;
  $s\leftarrow s + h(W+\tau W_o - W\circ W_o)$\;
}
\KwOut{$h'(W,O)=s$}
\end{algorithm}

\section{Unified Perspective: From Traces to Priors and Back}
\label{sec:unified}
% ==================================================================

The two regimes connect bidirectionally:

\paragraph{From traces to structure-learning priors.}
VPO-VI (Algorithm~\ref{alg:vpo_vi}) produces posterior marginals
$\hat\pi_{ij}=\Pr(i\succ j\mid D)\approx
\frac{1}{N}\sum_n\indic[i\succ_{h(U^{(n)})}j]$.
Thresholding these at a calibrated level yields a partial order $O$ that
can be injected into NOTEARS/DAGMA via Algorithm~\ref{alg:po_dsl},
avoiding the $\Theta(m^2)$ cost of na\"ive transitive-closure enumeration.

\paragraph{From domain priors to trace inference.}
Suppose an expert supplies partial-order constraints $O$ over actions
(e.g., ``approve before deploy'').
One might be tempted to enforce all implied constraints in $O^+$;
the following theorem shows this is unnecessary.

\begin{theorem}[Cover constraints suffice under dominance]
\label{thm:cover}
Let $O$ be a strict partial order on $[m]$ with transitive reduction
$O^-$.
If $U_{i,k}>U_{j,k}$ for all $k\in[K]$ and all $(i,j)\in O^-$,
then $i\succ_{\Dom(U)}j$ for every $(i,j)\in O^+$.
\end{theorem}
\begin{proof}
Take any $(i,j)\in O^+$.
There exists a directed path $i=v_0\to v_1\to\cdots\to v_r=j$
using edges in $O^-$.
For each edge $(v_{t-1},v_t)\in O^-$, the hypothesis gives
$U_{v_{t-1},k}>U_{v_t,k}$ for all $k$.
By transitivity of $>$ in $\R$,
$U_{i,k}>U_{j,k}$ for all $k$, hence $i\succ_{\Dom(U)}j$.
\end{proof}

Theorem~\ref{thm:cover} shows that enforcing dominance on the
$|O^-|$ cover edges (often $O(m)$) suffices.
In practice, we inject the prior as a soft log-penalty:
\begin{equation}
\log p(O\mid U)
= \lambda\sum_{(i,j)\in O^-}\sum_{k=1}^K
\log\sigma\!\Bigl(\frac{U_{i,k}-U_{j,k}}{\tau}\Bigr),
\label{eq:soft_prior}
\end{equation}
where $\sigma$ is the sigmoid and $\tau>0$ a temperature.
This term is added to $\log p(D,z)$ in Eq.~\eqref{eq:log_weight}
during VPO-VI, giving the maximal-path
philosophy of~\citet{ban2024differentiable} \emph{inside}
the embedding-based inference.

\section{Related Work}
\label{sec:related}
% ==================================================================

\paragraph{Poset inference from sequences.}
Process mining~\citep{vanderaalst2016process,leemans2013discovering,weijters2006process}
discovers workflow models from event logs, but typically outputs Petri nets
rather than explicit posets with uncertainty.
Bayesian approaches to learning partial orders from rankings exist in the
choice-modeling
literature~\citep{marden1996analyzing} but do not scale to the
trace-de-linearization setting we consider.

\paragraph{Variational inference and gradient estimation.}
The IWAE bound~\citep{burda2016iwae} tightens the ELBO with multiple
importance samples.
Score-function gradient estimators~\citep{williams1992reinforce} suffer
from high variance; VIMCO~\citep{mnih2016vimco} introduces leave-one-out
baselines, and doubly-reparameterized
estimators~\citep{tucker2019doubly} further reduce variance when partial
reparameterization is possible.
\citet{daudel2026importance} unify these in the VR-IWAE framework and
derive the VIMCO-$\star$ estimator with provable $\sqrt{N}$ SNR.
Our work is the first to specialize this estimator to latent-poset models.

\paragraph{Differentiable structure learning with order constraints.}
NOTEARS~\citep{zheng2018notears} and DAGMA~\citep{bello2022dagma} enable
continuous optimization over DAG space.
\citet{ban2024differentiable} propose augmented-acyclicity constraints
for partial-order priors, avoiding the quadratic cost of na\"ive
transitive-closure enumeration.
Our Theorem~\ref{thm:cover} complements their result by showing that
dominance-based embeddings propagate transitivity automatically,
enabling even more compact prior injection.

\section{Discussion and Limitations}
\label{sec:discussion}
% ==================================================================

\paragraph{When does VPO help most?}
VPO-VI is most beneficial when (i)~the number of actions $m$ is moderate
(up to a few hundred), so the $O(m^2 K)$ dominance computation remains
tractable, and (ii)~traces are informative but noisy, so uncertainty
quantification matters.
For very large $m$, amortized or hierarchical extensions may be needed.

\paragraph{Choice of $K$.}
The embedding dimension $K$ controls poset density.
We fix $K$ in the current formulation; extending to inference over $K$
(e.g., via reversible-jump MCMC~\citep{green1995reversible} or a
categorical variational factor) is a natural direction.

\paragraph{Relaxing the hard dominance map.}
A continuous relaxation of $\Dom(U)$ (e.g., via sigmoid-smoothed
comparisons) could enable partial reparameterization and potentially
better gradient estimates.
This would trade off exact poset semantics for smoother optimization
landscapes.

\paragraph{Broader impact.}
VPO is a general-purpose inference tool for partial orders with no
direct negative societal impact.
Applications in workflow optimization or causal discovery should be
deployed with standard caveats about confounding and data quality.

\section{Conclusion}
\label{sec:conclusion}
% ==================================================================

We presented Variational Partial Orders (VPO), a framework that unifies
latent-poset inference from traces and partial-order-constrained structure
learning.
Our approach combines a latent-embedding poset model with
importance-weighted variational inference, using the VIMCO-$\star$ gradient
estimator to circumvent the non-differentiability of the dominance map
while achieving $\sqrt{N}$ SNR scaling.
We proved that enforcing dominance constraints on cover relations alone
suffices to propagate the full transitive closure
(Theorem~\ref{thm:cover}), enabling compact prior injection that
complements the augmented-acyclicity
approach~\citep{ban2024differentiable}.
Experiments demonstrate competitive poset recovery with calibrated
uncertainty, graceful computational scaling, and improved downstream
causal discovery.

\section{Full Derivation: Score-Function Gradient of the VR-IWAE Objective}
\label{app:sf_derivation}

We derive Eq.~\eqref{eq:reinforce_grad} from first principles.
Let $g(\phi)\coloneqq\frac{1}{N}\sum_{n=1}^N w(z_n)^{1-\alpha}$ where
$z_{1:N}\stackrel{iid}{\sim}q_\phi$.
The objective is
$\mathcal{L}^{(\alpha)}_N=\frac{1}{1-\alpha}\E[\log g(\phi)]$.

\paragraph{Step 1: Log-derivative trick.}
\begin{equation}
\nabla_\psi\mathcal{L}^{(\alpha)}_N
= \frac{1}{1-\alpha}\,\E\!\left[\frac{\nabla_\psi g(\phi)}{g(\phi)}\right].
\end{equation}

\paragraph{Step 2: Expand $\nabla_\psi g$.}
We write
$g(\phi)=\frac{1}{N}\sum_n\int w(z_n)^{1-\alpha}\,q_\phi(z_n)\,dz_n$
(inside the expectation, $z_n$ is fixed; the gradient acts on $q_\phi$).
Using $w(z)^{1-\alpha}=p(D,z)^{1-\alpha}/q_\phi(z)^{1-\alpha}$
and the product rule:
\begin{align}
\nabla_\psi\bigl[w(z_n)^{1-\alpha}\,q_\phi(z_n)\bigr]
&= w(z_n)^{1-\alpha}\Bigl[
  (1\!-\!\alpha)\,\nabla_\psi\log w(z_n)
  + \nabla_\psi\log q_\phi(z_n)
\Bigr]\,q_\phi(z_n).
\end{align}
Hence:
\begin{equation}
\nabla_\psi g(\phi)
= \frac{1}{N}\sum_{n=1}^N w(z_n)^{1-\alpha}\Bigl[
  (1\!-\!\alpha)\,\nabla_\psi\log w(z_n)
  +\nabla_\psi\log q_\phi(z_n)
\Bigr].
\end{equation}

\paragraph{Step 3: Divide by $g$ and collect terms.}
With $\bar{W}_n^{(1-\alpha)}=w(z_n)^{1-\alpha}/\bigl(N\,g(\phi)\bigr)$:
\begin{align}
\frac{\nabla_\psi g}{g}
&= (1\!-\!\alpha)\sum_n\bar{W}_n^{(1-\alpha)}\nabla_\psi\log w(z_n)
  + \sum_n\bar{W}_n^{(1-\alpha)}\nabla_\psi\log q_\phi(z_n).
\end{align}
Multiplying by $1/(1-\alpha)$ and taking expectations yields
Eq.~\eqref{eq:reinforce_grad}.
The second term can be rewritten with a global $\log g$ signal via
the score-function identity $\E[\nabla_\psi\log q_\phi(z)]=0$.

\section{VIMCO-$\star$ Derivation and Variance Properties}
\label{app:vimco_star_derivation}

\paragraph{Control-variate principle.}
The problematic REINFORCE term in~\eqref{eq:reinforce_grad} multiplies
$\nabla_\psi\log q_\phi(z_n)$ by the global signal
$\ell\coloneqq\frac{1}{1-\alpha}\log g$.
For each sample $n$, we seek a baseline $b_n=b(z_{-n})$ such that
$\ell-b_n$ has lower variance while maintaining unbiasedness
(since $\E[\nabla_\psi\log q_\phi(z_n)\cdot b(z_{-n})]=0$).

\paragraph{VIMCO baseline (classical).}
\citet{mnih2016vimco} set $b_n^{\mathrm{VIMCO}}$ to the
log-mean-exp of leave-one-out weights with a geometric-mean replacement
for $z_n$.

\paragraph{VIMCO-$\star$ baseline ($\alpha=0$).}
\citet{daudel2026importance} show that for $\alpha=0$, the choice
$b_n^{\star}=\log(1-W_n)$ (where $W_n=w(z_n)/\sum_j w(z_j)$)
yields a simpler estimator (Eq.~\ref{eq:vimco_star}) with superior
variance properties.

\begin{proposition}[SNR scaling, {\citet[Theorem~3]{daudel2026importance}}]
Under mild regularity conditions on $q_\phi$ and $p$, the SNR of the
VIMCO-$\star$ estimator at $\alpha=0$ satisfies
$\SNR\bigl(\widehat{\nabla_\psi\mathcal{L}^{(0)}_N}\bigr)
=\Theta(\sqrt{N})$ as $N\to\infty$,
whereas the naive multi-sample REINFORCE estimator has
$\SNR=O(N^{-1/2})$.
\end{proposition}

\paragraph{Sign analysis.}
For $W_n\in(0,1)$, the per-sample coefficient in
Eq.~\eqref{eq:vimco_star_simple} satisfies
$W_n+\log(1-W_n)\le 0$
(with equality iff $W_n=0$), ensuring the effective signal is
non-negative and encourages $\phi$ to upweight high-importance particles.

\section{Proof of Theorem~\ref{thm:cover}}
\label{app:cover_proof}

We restate and prove the theorem for completeness.

\begin{theorem*}[Cover constraints suffice under dominance]
Let $O$ be a strict partial order on $[m]$ and $O^-$ its transitive
reduction.
If $U_{i,k}>U_{j,k}$ for all $k\in[K]$ and all $(i,j)\in O^-$,
then $i\succ_{\Dom(U)}j$ for every $(i,j)\in O^+$.
\end{theorem*}

\begin{proof}
Let $(i,j)\in O^+$.
By definition of the transitive closure, there exists a
path $i=v_0\to v_1\to\cdots\to v_r=j$ with each edge
$(v_{s-1},v_s)\in O^-$.
The hypothesis gives $U_{v_{s-1},k}>U_{v_s,k}$ for all $k$ and all $s$.
Chaining these $r$ strict inequalities:
$U_{i,k}=U_{v_0,k}>U_{v_1,k}>\cdots>U_{v_r,k}=U_{j,k}$
for every $k\in[K]$.
Hence $U_{i,k}>U_{j,k}$ for all $k$, which is exactly
$i\succ_{\Dom(U)}j$.
\end{proof}

\section{Complexity Analysis}
\label{app:complexity}

\paragraph{VPO-VI per-iteration cost.}
For each of $N$ particles:
(i)~building $\Dom(U^{(n)})\in\{0,1\}^{m\times m}$ costs $O(m^2 K)$
via broadcasted comparison;
(ii)~evaluating the trace likelihood for a single trace of length $T$
costs $O(Tm)$ when maintaining an indegree array for frontier updates.
Total: $O\bigl(N(m^2 K + n\bar{T}m)\bigr)$ where
$\bar{T}=\frac{1}{n}\sum_i T_i$.

\paragraph{PO-DSL constraint cost.}
The na\"ive penalty (Eq.~\ref{eq:naive_penalty}) requires
$|O^+|$ evaluations, each involving $O(d^3)$ matrix operations,
totaling $O(|O^+|\cdot d^3)$.
The augmented approach (Algorithm~\ref{alg:po_dsl}) requires
$|\mathcal{P}(O^-)|$ evaluations at $O(d^3)$ each.
For a chain, $|\mathcal{P}(O^-)|=1$ versus $|O^+|=\Theta(m^2)$.

\paragraph{Cover-constraint prior.}
Evaluating Eq.~\eqref{eq:soft_prior} costs $O(|O^-|\cdot K)$, which is
$O(mK)$ for chains---negligible relative to the dominance computation.

\section{Experimental Details}

\paragraph{Hyperparameters.}
All VPO-VI runs use Adam~\citep{kingma2014adam} with learning rate
$\eta=10^{-2}$, $N=16$ particles, gradient clipping at $\|\hat g\|=10$,

The variational log-standard-deviations are clipped to $[-5,2]$.
MCMC baselines use $5000$ samples with $1000$ burn-in.
